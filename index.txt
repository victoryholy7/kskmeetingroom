        async function apiCall(action, payload = {}) {
            const res = await fetch(API_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                body: JSON.stringify({ action, payload })
            });
            const data = await res.json();

        function addSelectedDate() {
            const dateInput = document.getElementById('bookingDate');
            const selectedDate = dateInput.value;
            
            if (!selectedDate) {
                alert('⚠️ กรุณาเลือกวันที่ก่อน');
                return;
            }
            
            if (!isWeekday(selectedDate)) {
                alert('⚠️ สามารถจองได้เฉพาะวันจันทร์ - ศุกร์ เท่านั้น');
                return;

                alert('⚠️ วันที่นี้ได้เลือกไว้แล้ว');
                return;
            }
            
            selectedDatesArray.push(selectedDate);
            updateSelectedDatesDisplay();
            dateInput.value = '';
        }

            if (selectedDatesArray.length === 0) {

                return;

                const dateStr = dateObj.toLocaleDateString('th-TH', {
                    weekday: 'short',
                    month: 'short',
                    day: 'numeric'
                });
                
                return `
                    <span class="inline-flex items-center bg-emerald-100 text-emerald-800 px-3 py-1 rounded-full text-sm">

                        <button type="button" onclick="removeSelectedDate('${date}')" class="ml-2 text-red-500 hover:text-red-700">×</button>

                `;

        async function login(identifier, password) {

                currentUser = result.user;

        async function register(idCard, password, name) {
            try {
                if (idCard.length !== 4 || !/^\d+$/.test(idCard)) throw new Error('กรุณาใส่เลขบัตรประชาชน 4 ตัวท้าย (ตัวเลขเท่านั้น)');
                if (password.length < 4) throw new Error('รหัสผ่านต้องมีอย่างน้อย 4 ตัวอักษร');
                if (name.trim().length < 2) throw new Error('กรุณาใส่ชื่อ-นามสกุลให้ครบถ้วน');
                await apiCall('register', { idCard, password, name: name.trim() });
                showSuccessPopup('🎉 สมัครสมาชิกสำเร็จ!', `ยินดีต้อนรับ ${name} สามารถเข้าสู่ระบบได้แล้ว`);
                closeRegisterModal();
            } catch (error) {
                alert('❌ ' + error.message);
            }
        }

        async function changePassword(idCard, newPassword, confirmPassword) {
            try {
                if (idCard.length !== 4 || !/^\d+$/.test(idCard)) throw new Error('กรุณาใส่เลขบัตรประชาชน 4 ตัวท้าย (ตัวเลขเท่านั้น)');
                if (newPassword.length < 4) throw new Error('รหัสผ่านใหม่ต้องมีอย่างน้อย 4 ตัวอักษร');
                if (newPassword !== confirmPassword) throw new Error('รหัสผ่านใหม่และยืนยันรหัสผ่านไม่ตรงกัน');
                await apiCall('changePassword', { idCard, newPassword });
                showSuccessPopup('🎉 เปลี่ยนรหัสผ่านสำเร็จ!', 'รหัสผ่านของคุณได้ถูกเปลี่ยนเรียบร้อยแล้ว');
                closeForgotPasswordModal();
            } catch (error) {
                alert('❌ ' + error.message);
            }
        }

        async function loadBookingHistory() {
            try {
                const result = await apiCall('listHistory', {});
                const allBookings = result.history;
                const roomNames = { 'room1': 'ห้องประชุมอิ่มใจ', 'room2': 'ห้องประชุมอิ่มสุข', 'room3': 'ห้องประชุมเนตรประชา', 'room4': 'Working Space' };
                const historyContainer = document.getElementById('bookingHistory');
                if (allBookings.length === 0) {
                    historyContainer.innerHTML = '<p class="text-gray-500 text-center">ยังไม่มีประวัติการจอง</p>';
                    return;
                }
                const groupedByMonth = {};
                allBookings.forEach(booking => {
                    const date = new Date(booking.date);
                    const monthKey = `${date.getFullYear()}-${String(date.getMonth()).padStart(2, '0')}`;
                    const monthName = date.toLocaleDateString('th-TH', { year: 'numeric', month: 'long' });
                    if (!groupedByMonth[monthKey]) groupedByMonth[monthKey] = { name: monthName, bookings: [], sortKey: monthKey };
                    groupedByMonth[monthKey].bookings.push(booking);
                });
                const sortedMonths = Object.values(groupedByMonth)
                    .sort((a, b) => b.sortKey.localeCompare(a.sortKey))
                    .map(month => { month.bookings.sort((a, b) => new Date(a.date) - new Date(b.date)); return month; });

                historyContainer.innerHTML = `
                    <div class="grid gap-6">
                        ${sortedMonths.map(month => `
                            <div class="border rounded-lg p-6 bg-gradient-to-br from-purple-50 to-blue-50 shadow-lg relative">
                                <button onclick="deleteMonthHistory('${month.sortKey}')" 
                                        class="absolute top-4 right-4 text-red-500 hover:text-red-700 text-xl transition-colors duration-200 hover:bg-red-100 rounded-full w-8 h-8 flex items-center justify-center">×</button>
                                <h4 class="font-bold text-purple-700 mb-6 text-xl flex items-center">
                                    📅 ${month.name}
                                    <span class="ml-3 bg-purple-200 text-purple-800 px-3 py-1 rounded-full text-sm">${month.bookings.length} รายการ</span>
                                </h4>
                                <div class="grid md:grid-cols-2 lg:grid-cols-3 gap-4">
                                    ${month.bookings.map(booking => {
                                        const bookingDate = new Date(booking.date);
                                        const dayName = bookingDate.toLocaleDateString('th-TH', { weekday: 'short' });
                                        const dayNumber = bookingDate.getDate();
                                        let roomBgColor, roomTextColor, roomBorderColor, roomDateBg, roomDateText;
                                        switch (booking.room) {
                                            case 'room1': roomBgColor = 'bg-gradient-to-br from-cyan-50 to-white'; roomTextColor = 'text-cyan-800'; roomBorderColor = 'border-l-4 border-cyan-400'; roomDateBg = 'bg-cyan-100'; roomDateText = 'text-cyan-600'; break;
                                            case 'room2': roomBgColor = 'bg-gradient-to-br from-teal-50 to-white'; roomTextColor = 'text-teal-800'; roomBorderColor = 'border-l-4 border-teal-400'; roomDateBg = 'bg-teal-100'; roomDateText = 'text-teal-600'; break;
                                            case 'room3': roomBgColor = 'bg-gradient-to-br from-emerald-50 to-white'; roomTextColor = 'text-emerald-800'; roomBorderColor = 'border-l-4 border-emerald-500'; roomDateBg = 'bg-emerald-100'; roomDateText = 'text-emerald-600'; break;
                                            case 'room4': roomBgColor = 'bg-gradient-to-br from-pink-50 to-white'; roomTextColor = 'text-pink-800'; roomBorderColor = 'border-l-4 border-pink-400'; roomDateBg = 'bg-pink-100'; roomDateText = 'text-pink-600'; break;
                                            default: roomBgColor = 'bg-white'; roomTextColor = 'text-gray-800'; roomBorderColor = 'border-l-4 border-purple-400'; roomDateBg = 'bg-purple-100'; roomDateText = 'text-purple-600';
                                        }
                                        const uniqueId = booking.id;
                                        return `
                                            <div class="${roomBgColor} rounded-lg p-4 ${roomBorderColor} shadow-md hover:shadow-lg transition-shadow cursor-pointer relative" onclick='showBookingDetail(${JSON.stringify(booking)})'>
                                                <button onclick="event.stopPropagation(); deleteBookingFromHistory('${uniqueId}')" 
                                                        class="absolute top-2 right-2 text-red-500 hover:text-red-700 text-lg transition-colors duration-200 hover:bg-red-100 rounded-full w-6 h-6 flex items-center justify-center">×</button>
                                                <div class="flex items-start space-x-3">
                                                    <div class="${roomDateBg} rounded-lg p-2 text-center min-w-[50px]">
                                                        <div class="text-xs ${roomDateText} font-medium">${dayName}</div>
                                                        <div class="text-lg font-bold ${roomTextColor}">${dayNumber}</div>
                                                    </div>
                                                    <div class="flex-1 min-w-0 pr-6">
                                                        <p class="text-sm ${roomTextColor} flex items-center mt-1 font-medium">🏢 ${roomNames[booking.room]}</p>
                                                        <p class="text-xs ${roomTextColor} opacity-75 mt-1">คลิกเพื่อดูรายละเอียด</p>
                                                    </div>
                                                </div>
                                            </div>`;
                                    }).join('')}
                                </div>
                            </div>`).join('')}
                    </div>`;
            } catch (error) {
                console.error('Error loading booking history:', error);
            }
        }

            const startTimeSelect = document.getElementById('bookingStartTime');

            const bookerNameInput = document.getElementById('bookerName');
            const phoneInput = document.getElementById('bookerPhone');
            const workGroupSelect = document.getElementById('workGroup');
            const topicTextarea = document.getElementById('meetingTopic');

            const hasSelectedDates = selectedDatesArray.length > 0;
            const hasSingleDate = dateInput.value.trim();
            if (!hasSelectedDates && !hasSingleDate) { alert('⚠️ กรุณาเลือกวันที่อย่างน้อย 1 วัน'); return; }

            let hasError = false;
            const requiredFields = [roomSelect, startTimeSelect, endTimeSelect, bookerNameInput, phoneInput, workGroupSelect, topicTextarea];
            if (!hasSelectedDates) requiredFields.push(dateInput);
            requiredFields.forEach(field => {
                if (!field.value.trim()) {
                    hasError = true;
                    const errorIcon = document.createElement('span');
                    errorIcon.className = 'error-indicator text-red-500 text-xl absolute right-3 top-1/2 transform -translate-y-1/2';
                    errorIcon.innerHTML = '❌';
                    field.parentElement.style.position = 'relative';
                    field.parentElement.appendChild(errorIcon);
                }
            });
            if (hasError) return;

            const equipment = Array.from(document.querySelectorAll('input[type="checkbox"]:checked')).map(cb => cb.value);
            const bookingData = {
                room: roomSelect.value, date: dateInput.value,
                startTime: startTimeSelect.value, endTime: endTimeSelect.value,
                equipment: equipment, bookerName: bookerNameInput.value,
                phone: phoneInput.value, workGroup: workGroupSelect.value, topic: topicTextarea.value
            };
            if (bookingData.startTime >= bookingData.endTime) { alert('⚠️ เวลาเริ่มต้นต้องน้อยกว่าเวลาสิ้นสุด'); return; }
            createBooking(bookingData);
        });

        function showDeleteConfirmModal(bookingId, isMonth = false, monthKey = null) {
            const popup = document.createElement('div');
            popup.className = 'fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50';
            popup.innerHTML = `
                <div class="bg-white rounded-2xl p-8 max-w-md w-full mx-4 shadow-2xl">
                    <div class="text-center">
                        <div class="bg-red-100 w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4">
                            <span class="text-2xl text-red-600">🗑️</span>
                        </div>
                        <h3 class="text-xl font-bold text-gray-800 mb-2">ยืนยันการลบ</h3>
                        <p class="text-gray-600 mb-6">${isMonth ? 'คุณต้องการลบประวัติทั้งเดือนนี้หรือไม่?' : 'คุณต้องการลบรายการนี้ออกจากประวัติหรือไม่?'}</p>
                        <div class="flex space-x-4">
                            <button onclick="confirmDelete('${bookingId}', ${isMonth}, '${monthKey}', this.parentElement.parentElement.parentElement.parentElement)" 
                                    class="flex-1 bg-red-500 hover:bg-red-600 text-white py-3 rounded-xl transition-all duration-300 font-medium">ยืนยัน</button>
                            <button onclick="this.parentElement.parentElement.parentElement.parentElement.remove()" 
                                    class="flex-1 bg-gray-500 hover:bg-gray-600 text-white py-3 rounded-xl transition-all duration-300 font-medium">ยกเลิก</button>
                        </div>
                    </div>
                </div>`;
            document.body.appendChild(popup);
        }

        async function confirmDelete(bookingId, isMonth, monthKey, popup) {
            try {
                if (isMonth) {
                    await apiCall('deleteMonthHistory', { monthKey });
                } else {
                    await apiCall('deleteHistoryItem', { bookingId });
                }
                popup.remove();
                await loadBookings();
                showSuccessPopup('✅ ลบสำเร็จ!', 'ข้อมูลได้ถูกลบออกจากระบบเรียบร้อยแล้ว');
            } catch (error) {
                alert('❌ เกิดข้อผิดพลาด: ' + error.message);
                popup.remove();
            }
        }

        async function deleteBookingFromHistory(bookingId) { showDeleteConfirmModal(bookingId, false); }
        async function deleteMonthHistory(monthKey) { showDeleteConfirmModal('', true, monthKey); }
